import random
import string

def generate_password(length=12, include_letters=True, include_numbers=True, include_special_chars=True):
characters = ”
if include_letters:
characters += string.ascii_letters
if include_numbers:
characters += string.digits
if include_special_chars:
characters += string.punctuation

password = ”.join(random.choice(characters) for _ in range(length))
return password

# Example usage
length = int(input(“Enter the desired password length: “))
include_letters = input(“Include letters? (y/n): “).lower() == ‘y’
include_numbers = input(“Include numbers? (y/n): “).lower() == ‘y’
include_special_chars = input(“Include special characters? (y/n): “).lower() == ‘y’

password = generate_password(length, include_letters, include_numbers, include_special_chars)
print(“Generated Password:”, password)

This code allows users to specify the desired password length and select whether to include letters, numbers, and special characters. The generate_password function generates a random password based on the selected parameters. You can further enhance this code by adding a user interface and implementing secure storage mechanisms to store the generated passwords, such as using a password manager or encrypting the stored passwords.
